Datastores

  • Use on the server only
  • Save player coins, items, etc.
local ds = game:GetService("DataStoreService"):GetDataStore("Coins")

game.Players.PlayerAdded:Connect(function(player)
  local data = ds:GetAsync(player.UserId) or 0
  print("Coins:", data)
end)

game.Players.PlayerRemoving:Connect(function(player)
  ds:SetAsync(player.UserId, 100) -- Example save
end)
